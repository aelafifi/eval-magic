export function tryCatchSeq(...fns: (() => any)[]) {
  if (fns.length > 0) {
    const [fn, ...rest] = fns;
    try {
      return fn();
    } catch (e) {
      if (rest.length > 0) {
        return tryCatchSeq(...rest);
      }
      throw e;
    }
  }
}
